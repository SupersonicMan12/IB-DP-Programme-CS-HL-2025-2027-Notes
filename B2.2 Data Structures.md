Stack (using arrays)
```java
class stack{
	private int[] a = new int[10];
	private int idx;
	public stack(){}
	public void push(int x){
		a[++idx] = x;
	}
	public void pop(){
		idx--;
	}
	public int size(){
		return idx;
	}
	public boolean isEmpty(){
		return (idx == 0);
	}
	public int peek(){
		return a[idx];
	}
}
```

Queue (using arrays)
```java
class queue{
	private int[] a = new int[100];
	private int front, back;
	public queue(){
		front = 0, back = 0;
	}
	public void add(int x){
		a[++front] = x;
	}
	public void pop(){
		back++;
	}
	public int size(){
		return (front-back);
	}
	public boolean isEmpty(){
		return (front-back == 0);
	}
	public int peek(){
		return a[back];
	}
}
```
